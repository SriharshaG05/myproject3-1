pipeline {
    agent any

    tools {
        maven "maven"
    }

    stages {
        stage('Build') {
            steps {
                git 'https://github.com/jglick/simple-maven-project-with-tests.git'
                bat "mvn -Dmaven.test.failure.ignore=true clean package"
            }
        }
    }

    post {
        // Always publish test results & jar, even if build is UNSTABLE
        always {
            junit '**/target/surefire-reports/TEST-*.xml'
            archiveArtifacts artifacts: 'target/*.jar', fingerprint: true
        }
    }
}

# Start Minikube using Docker Desktop
minikube start --driver=docker

# Check Minikube status
minikube status

# Create Nginx deployment
kubectl create deployment my-nginx --image=nginx

# Check pods
kubectl get pods

# Expose deployment as NodePort
kubectl expose deployment my-nginx --type=NodePort --port=80 --name=nginx-service

# Change NodePort to 8090
kubectl patch service nginx-service -p '{"spec": {"ports": [{"port":80, "nodePort":8090}]}}'

# Check services
kubectl get svc

# Get service URL
minikube service nginx-service --url

# (Open in browser: http://localhost:8090)

# List pods and services
kubectl get pods
kubectl get svc

# Stop Minikube (optional)
minikube stop

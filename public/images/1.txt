Step 1: Start Minikube
------------------------------
minikube start

Step 2: Create Redis Deployment
------------------------------
kubectl create deployment redis-deploy --image=redis:latest

Step 3: Expose Redis on port 6379
------------------------------
kubectl expose deployment redis-deploy --type=NodePort --port=6379

Step 4: Verify
------------------------------
kubectl get pods
kubectl get svc

Step 5: Access Redis using port-forward
------------------------------
kubectl port-forward svc/redis-deploy 6379:6379

Step 6: Test Redis (optional)
------------------------------
redis-cli -h 127.0.0.1 -p 6379
set name harsha
get name

Cleanup (Optional)
------------------------------
kubectl delete deployment redis-deploy
kubectl delete service redis-deploy

pipeline {

    agent any

    }

    stages {

        stage('git repo & clean') {
            steps {
                // Clone GitHub project

                // Clean the project using Maven

            }
        }

        stage('install') {
            steps {
                // Install dependencies
                bat "mvn install -f mavenjava"
            }
        }

        stage('test') {
            steps {
                // Run Test Cases
                bat "mvn test -f mavenjava"
            }
        }

        stage('package') {
            steps {
                // Package JAR/WAR
                bat "mvn package -f mavenjava"
            }
        }

    }
}
